00062         {

00064 
00065           float tmp = -_reference + sqrt(output.real()*output.real() +
00066                                          output.imag()*output.imag());

00068           if((tmp) > _gain) {
00069             rate = _attack_rate;

00071           _gain -= tmp*rate;
